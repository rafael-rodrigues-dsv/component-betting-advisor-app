# domain/services/impl/value_bet_calculator_impl.py

class ValueBetCalculatorImpl(ValueBetCalculatorContract):
    """Calcula value bets comparando probabilidades com odds"""
    
    MIN_VALUE_THRESHOLD = 0.05  # 5%
    MIN_CONFIDENCE_THRESHOLD = 0.50  # 50%
    
    def calculate(
        self, 
        ensemble_result: EnsembleResult, 
        odds: Dict[str, float]
    ) -> List[ValueBetResult]:
        
        results = []
        
        markets = [
            ("home_win", ensemble_result.home_win, odds.get("home")),
            ("draw", ensemble_result.draw, odds.get("draw")),
            ("away_win", ensemble_result.away_win, odds.get("away")),
            ("over_25", ensemble_result.over_25, odds.get("over_25")),
            ("btts_yes", ensemble_result.btts_yes, odds.get("btts_yes")),
        ]
        
        for market, real_prob, odd in markets:
            if odd is None:
                continue
                
            implied_prob = 1 / odd
            value = real_prob - implied_prob
            expected_value = (real_prob * odd) - 1
            
            recommendation = self._get_recommendation(value, real_prob)
            
            results.append(ValueBetResult(
                market=market,
                real_probability=real_prob,
                implied_probability=implied_prob,
                odd=odd,
                value=value,
                expected_value=expected_value,
                recommendation=recommendation
            ))
        
        return sorted(results, key=lambda x: x.value, reverse=True)
    
    def _get_recommendation(self, value: float, confidence: float) -> str:
        if value > 0.10 and confidence > 0.60:
            return "STRONG_BET"
        elif value > 0.05 and confidence > 0.50:
            return "VALUE_BET"
        elif value > 0.05:
            return "RISKY"
        return "SKIP"
